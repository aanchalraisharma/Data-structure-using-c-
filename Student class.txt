#include <iostream>
#include <string>
using namespace std;

class Student {
private:
    int rollNo;
    float marks;

public:
    string collegeName;   // public data member

    Student(int r, float m, string c) {
        rollNo = r;
        marks = m;
        collegeName = c;
    }

    void display() {
        cout << "Roll No: " << rollNo << endl;
        cout << "Marks: " << marks << endl;
        cout << "College Name: " << collegeName << endl;
        cout << "------------------------" << endl;
    }
};

int main() {
    Student s1(101, 85.5, "ABC College");
    Student s2(102, 90.0, "ABC College");
    Student s3(103, 78.25, "XYZ Institute");

    s1.display();
    s2.display();
    s3.display();

    return 0;
}
